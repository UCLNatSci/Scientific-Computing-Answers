import numpy as np
import matplotlib.pyplot as plt

s = 93 # launch speed 93 m/s
theta = np.pi/4 # pi/4 radians equals 45 degrees

DELTA_T = 0.01
g = 9.8

x = 0 # initial x, y positions
y = 0
vx = s * np.cos(theta) # initial x, y velocities
vy = s * np.sin(theta)

dis = 900

x_list = []
y_list = []

while y >= 0:
    x_list.append(x)
    y_list.append(y)
    x = x + vx * DELTA_T
    y = y + vy * DELTA_T
    
    vy = vy - g * DELTA_T
    
plt.figure(figsize=(4,4))
plt.plot(x_list, y_list)
plt.xlim(0, 1000)
plt.ylim(0, 1000)
plt.scatter(dis, 0) # plot the location of the target

if x > dis - 20 and x < dis + 20:
    print("hit")
else:
    print("miss")
